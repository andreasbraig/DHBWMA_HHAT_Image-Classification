	\vspace*{0\p@}%
	{\parindent \z@ \raggedright
		\normalfont
		\interlinepenalty\@M
		\fontsize{16pt}{16pt}\selectfont \bfseries \thechapter.\  #1\par\nobreak
		\vskip 10\p@
}}

	{-2.5ex \@plus -1ex \@minus -.2ex}% Abstand vor der Überschrift
